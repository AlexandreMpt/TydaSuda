        return {
            activeButton: 'walking'
        };
    },
    mounted() {
        // Инициализация: показываем начально активный контент
        this.showContent(this.activeButton);
        
        // Обработчики событий для переключения контента по клику на кнопки
        const buttons = document.querySelectorAll('.services__navbar__btn');
        buttons.forEach(button => {
            button.addEventListener('click', (event) => {
                event.preventDefault();
                const target = button.getAttribute('data-target');
                this.showContent(target);
                this.updateActiveButton(button);
            });
        });
    },
    methods: {
        showContent(target) {
            // Скрываем все блоки контента
            document.querySelectorAll('.services__block').forEach(block => {
                block.style.display = 'none';
            });
            // Показываем выбранный блок контента
            document.getElementById(`${target}Content`).style.display = 'block';
            // Обновляем активную кнопку
            this.activeButton = target;
        },
        updateActiveButton(button) {
            // Снимаем класс активной кнопки со всех кнопок
            document.querySelectorAll('.services__navbar__btn').forEach(btn => {
                btn.classList.remove('active-btn');
            });
            // Добавляем класс активной кнопки к выбранной кнопке
            button.classList.add('active-btn');
        }
    }
};

</script>

      this.gender = '';
      this.password = '';
      this.confirmPassword = '';

      this.isRegistrationVisible = false;

      this.isBurgerMenuOpen = !this.isBurgerMenuOpen; // Переключаем состояние бургер-меню
    }

    // Инициализация слушателя для открытия модального окна из подвала
